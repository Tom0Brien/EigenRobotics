28 template <typename Scalar, int nq>
29 Eigen::Transform<Scalar, 3, Eigen::Isometry> forward_kinematics(const Model<Scalar, nq>& model,
30 const Eigen::Matrix<Scalar, nq, 1>& q,
31 const int& target_link_idx,
32 const bool inverse_transform = false) {
33 // Get the target link
34 Link<Scalar> current_link = model.links[target_link_idx];
35
36 // Check if the link is within the kinematic tree
37 if (current_link.idx == -1) {
38 std::string error_msg = "Error: Link " + std::to_string(target_link_idx) + " not found.";
39 throw std::runtime_error(error_msg);
40 }
41
42 // Return identity transform if the link is itself the base link
43 if (current_link.idx == model.base_link_idx) {
44 return Eigen::Transform<Scalar, 3, Eigen::Isometry>::Identity();
45 }
46
47 // Build kinematic tree from target_link {t} to base {b}
48 Eigen::Transform<Scalar, 3, Eigen::Isometry> Htb = Eigen::Transform<Scalar, 3, Eigen::Isometry>::Identity();
49 while (current_link.idx != model.links[model.base_link_idx].idx) {
50 Eigen::Transform<Scalar, 3, Eigen::Isometry> H = Eigen::Transform<Scalar, 3, Eigen::Isometry>::Identity();
51 auto current_joint = current_link.joint;
52 if (current_joint.type == JointType::REVOLUTE) {
53 // Rotate by q_current around axis
54 H.linear() =
55 Eigen::AngleAxis<Scalar>(Scalar(q(current_joint.idx)), current_joint.axis).toRotationMatrix();
56 }
57 else if (current_joint.type == JointType::PRISMATIC) {
58 // Translate by q_current along axis
59 H.translation() = current_joint.axis * Scalar(q(current_joint.idx));
60 }
61 Htb = Htb * H.inverse();
62 // Apply inverse joint transform as we are going back up tree
63 Htb = Htb * current_joint.parent_transform.inverse();
64 // Move up the tree to parent link
65 current_link = model.links[current_link.parent];
66 }
67
68 // If inverse transform is requested, return Hbt, which is transform from target {t} to base {b}
69 if (inverse_transform) {
70 return Htb;
71 }
72 // Return transform Hbt, which is transform from base {b} to target {t}
73 return Htb.inverse();
74 }

106 template <typename Scalar, int nq>
107 Eigen::Transform<Scalar, 3, Eigen::Isometry> forward_kinematics(const Model<Scalar, nq>& model,
108 const Eigen::Matrix<Scalar, nq, 1>& q,
109 const std::string& source_link_name,
110 const std::string& target_link_name) {
111 // Build kinematic tree from source {s} to base {b} frame
112 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hsb = forward_kinematics(model, q, source_link_name, true);
113 // Build kinematic tree from base {b} to target {t} frame
114 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbt = forward_kinematics(model, q, target_link_name);
115 // Compute transform between source {s} and target {t} frames
116 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hst = Hsb * Hbt;
117 return Hst;
118 }

262 template <typename Scalar, int nq>
263 Eigen::Matrix<Scalar, 6, nq> geometric_jacobian(const Model<Scalar, nq>& model,
264 const Eigen::Matrix<Scalar, nq, 1>& q,
265 const int& target_link_idx) {
266 // Initialize the geometric jabobian matrix with zeros
267 Eigen::Matrix<Scalar, 6, nq> J = Eigen::Matrix<Scalar, 6, nq>::Zero();
268
269 // Get the target link from the model
270 Link<Scalar> current_link = model.links[target_link_idx];
271
272 // Get the base link from the model
273 auto base_link = model.links[model.base_link_idx];
274
275 // Compute the displacement of the target link {t} in the base link frame {b}
276 Eigen::Matrix<Scalar, 3, 1> rTBb = translation(model, q, base_link.idx, current_link.idx);
277
278 while (current_link.idx != model.base_link_idx) {
279 auto current_joint = current_link.joint;
280 if (current_joint.idx != -1) {
281 // Compute the transform between base {b} and the current link {i}
282 auto Hbi = forward_kinematics(model, q, base_link.idx, current_link.idx);
283 // Axis of the current joint rotated into the base frame {b}
284 auto zIBb = Hbi.linear() * current_joint.axis;
285 // Compute the displacement of the current link {i} from the base link frame {b}
286 auto rIBb = Hbi.translation();
287 if (current_joint.type == JointType::PRISMATIC) {
288 J.block(0, current_joint.idx, 3, 1) = zIBb;
289 J.block(3, current_joint.idx, 3, 1) = Eigen::Matrix<Scalar, 3, 1>::Zero();
290 }
291 else if (current_joint.type == JointType::REVOLUTE) {
292 J.block(0, current_joint.idx, 3, 1) = (zIBb).cross(rTBb - rIBb);
293 J.block(3, current_joint.idx, 3, 1) = zIBb;
294 }
295 }
296 // Move up the tree to parent towards the base
297 current_link = model.links[current_link.parent];
298 }
299 return J;
300 }

379 template <typename Scalar, int nq>
380 Eigen::Matrix<Scalar, 3, 1> centre_of_mass(const Model<Scalar, nq>& model,
381 const Eigen::Matrix<Scalar, nq, 1>& q,
382 const std::string& source_link_name) {
383 // For each link in the model, compute the transform from the source link to the CoM of the link
384 Scalar total_mass = 0;
385 Eigen::Matrix<Scalar, 3, 1> rISs = Eigen::Matrix<Scalar, 3, 1>::Zero();
386 for (auto link : model.links) {
387 // Compute the transform from the source link to the CoM of the link
388 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hsc =
389 forward_kinematics_com(model, q, source_link_name, link->name);
390 // Compute the centre of mass of the link
391 rISs = rISs + Hsc.translation() * link->mass;
392 // Add the links mass to the total mass
393 total_mass += link->mass;
394 }
395 // Compute the centre of mass from the source link
396 return rISs / total_mass;
397 }
