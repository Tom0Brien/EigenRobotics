22 template <typename Scalar, int nq>
23 Eigen::Matrix<Scalar, nq, nq> mass_matrix(Model<Scalar, nq>& model, const Eigen::Matrix<Scalar, nq, 1>& q) {
24 // Reset the mass matrix and potential energy
25 model.data.M.setZero();
26 model.data.V = 0;
27 Eigen::Matrix<Scalar, 6, 6> Mi;
28 // Get the base link from the model
29 auto base_link = model.links[model.base_link_idx];
30 for (int i = 0; i < model.n_links; i++) {
31 Mi.setZero();
32 // Compute the rotation between the ith link and the base
33 Eigen::Matrix<Scalar, 3, 3> R0i = rotation(model, q, base_link.name, model.links[i].name);
34 // Insert the mass of the link into the top 3 diagonals
35 Mi.block(0, 0, 3, 3) = model.links[i].mass * Eigen::Matrix<Scalar, 3, 3>::Identity();
36 // Insert the inertia of the link into the bottom 3 diagonals
37 Mi.block(3, 3, 3, 3) = R0i * model.links[i].inertia * R0i.transpose();
38 // Compute the geometric jacobian of the links center of mass with respect to the base
39 Eigen::Matrix<Scalar, 6, nq> Jci = geometric_jacobian_com(model, q, model.links[i].name);
40 // Compute the contribution to the mass matrix of the link
41 model.data.M += Jci.transpose() * Mi * Jci;
42 // Compute the contribution to the potential energy of the link
43 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbi_c =
44 forward_kinematics_com<Scalar, nq>(model, q, model.base_link_idx, model.links[i].idx);
45 Eigen::Matrix<Scalar, 3, 1> rMIi_c = Hbi_c.translation();
46 model.data.V += -model.links[i].mass * model.gravity.transpose() * rMIi_c;
47 }
48 return model.data.M;
49 }

93 template <typename Scalar, int nq>
94 Eigen::Matrix<Scalar, 1, 1> hamiltonian(Model<Scalar, nq>& model,
95 const Eigen::Matrix<Scalar, nq, 1>& q,
96 const Eigen::Matrix<Scalar, nq, 1>& p) {
97 // Compute the mass matrix and potential energy
98 mass_matrix<Scalar, nq>(model, q);
99
100 // Compute the inverse of the mass matrix
101 Eigen::Matrix<Scalar, nq, nq> b = Eigen::Matrix<Scalar, nq, nq>::Identity();
102 Eigen::Matrix<Scalar, nq, nq> Minv = model.data.M.ldlt().solve(b);
103 model.data.Minv = Minv;
104
105 // Compute the kinetic energy
106 model.data.T = Scalar(0.5 * p.transpose() * Minv * p);
107
108 // Compute the total energy
109 Eigen::Matrix<Scalar, 1, 1> H = Eigen::Matrix<Scalar, 1, 1>(model.data.T + model.data.V);
110 return H;
111 }

121 template <typename Scalar, int nq, int ni>
122 Eigen::Matrix<Scalar, nq + nq, 1> forward_dynamics_without_constraints(Model<Scalar, nq>& model,
123 const Eigen::Matrix<Scalar, nq, 1>& q,
124 const Eigen::Matrix<Scalar, nq, 1>& p,
125 const Eigen::Matrix<Scalar, ni, 1>& u) {
126 // Number of states
127 const int nx = nq + nq;
128 // Cast to autodiff type for automatic differentiation
129 Eigen::Matrix<autodiff::real, nq, 1> q_ad(q); // the input vector q
130 Eigen::Matrix<autodiff::real, nq, 1> p_ad(p); // the input vector p
131 auto model_ad = model.template cast<autodiff::real>();
132
133 // Compute the jacobian of the hamiltonian wrt q and p
134 Eigen::Matrix<autodiff::real, 1, 1> H_ad;
135
136
137 Eigen::Matrix<Scalar, 1, nq> dH_dq =
138 autodiff::jacobian(hamiltonian<autodiff::real, nq>, wrt(q_ad), at(model_ad, q_ad, p_ad), H_ad);
139
140 Eigen::Matrix<Scalar, 1, nq> dH_dp = (model_ad.data.Minv * p_ad).template cast<Scalar>();
141
142 // Create the interconnection and damping matrix
143 Eigen::Matrix<Scalar, nx, nx> J = Eigen::Matrix<Scalar, nx, nx>::Zero();
144 J.block(0, nq, nq, nq) = Eigen::Matrix<Scalar, nq, nq>::Identity();
145 J.block(nq, 0, nq, nq) = -1 * Eigen::Matrix<Scalar, nq, nq>::Identity();
146 J.block(nq, nq, nq, nq) = model.data.Dp;
147
148 // Stack the jacobians of the hamiltonian
149 Eigen::Matrix<Scalar, nx, 1> dH = Eigen::Matrix<Scalar, nx, 1>::Zero();
150 dH.block(0, 0, nq, 1) = dH_dq.transpose();
151 dH.block(nq, 0, nq, 1) = dH_dp.transpose();
152
153 // Compute the forward dynamics: dx_dt
154 Eigen::Matrix<Scalar, nx, ni> G = Eigen::Matrix<Scalar, nx, ni>::Zero();
155 G.block(nq, 0, nq, nq) = model.data.Gp;
156 Eigen::Matrix<Scalar, nx, 1> dx_dt = J * dH + G * u;
157
158 // Store the result
159 model.data.dx_dt = dx_dt;
160 return dx_dt;
161 }

170 template <typename Scalar, int nq>
171 Eigen::Matrix<Scalar, 1, 1> hamiltonian_with_constraints(Model<Scalar, nq>& model,
172 const Eigen::Matrix<Scalar, nq, 1>& q,
173 const Eigen::Matrix<Scalar, nq, 1>& p,
174 const std::vector<std::string>& active_constraints) {
175 // Compute the mass matrix and potential energy for unconstrained dynamics
176 mass_matrix<Scalar, nq>(model, q);
177
178 // Compute the jacobian of the active constraints
179 Eigen::Matrix<Scalar, Eigen::Dynamic, nq> Jc;
180 for (int i = 0; i < active_constraints.size(); i++) {
181 // Compute the jacobian of the active constraint
182 auto Jci = geometric_jacobian(model, q, active_constraints[i]);
183 auto Jci_v = Jci.block(0, 0, 3, nq);
184 // Vertically Concatenate the jacobian of the active constraint to the jacobian of the active constraints
185 Jc.conservativeResize(Jc.rows() + Jci_v.rows(), nq);
186 Jc.block(Jc.rows() - Jci_v.rows(), 0, Jci_v.rows(), Jci_v.cols()) = Jci_v;
187 }
188 model.data.Jc.resize(Jc.rows(), Jc.cols());
189 model.data.Jc = Jc;
190
191 // Compute the left annihilator of the jacobian of the active constraints
192 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Jcp = null<Scalar>(Jc);
193 model.data.Jcp.resize(Jcp.rows(), Jcp.cols());
194 model.data.Jcp = Jcp;
195
196 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Gr = Jcp.transpose() * model.data.Gp;
197 model.data.Gr.resize(Gr.rows(), Gr.cols());
198 model.data.Gr = Gr;
199
200 model.data.Dp.setZero();
201 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Dr = Jcp.transpose() * model.data.Dp * Jcp;
202 model.data.Dr.resize(Dr.rows(), Dr.cols());
203 model.data.Dr = Dr;
204
205 auto Mr = Jcp.transpose() * model.data.M * Jcp;
206 model.data.Mr.resize(Mr.rows(), Mr.cols());
207 model.data.Mr = Mr;
208
209 model.data.nz = model.data.Gp.rows() - Gr.rows();
210 model.data.nr = Mr.rows();
211
212 // Compute the inverse of the mass matrix
213 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> b;
214 b.resize(model.data.nr, model.data.nr);
215 b.setIdentity();
216 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Mrinv = Mr.ldlt().solve(b);
217 model.data.Mrinv.resize(model.data.nr, model.data.nr);
218 model.data.Mrinv = Mrinv;
219
220 // Compute the kinetic energy
221 auto pr = p.tail(model.data.nr);
222 model.data.T = Scalar(0.5 * pr.transpose() * Mrinv * pr);
223
224 // Compute the total energy
225 Eigen::Matrix<Scalar, 1, 1> H = Eigen::Matrix<Scalar, 1, 1>(model.data.T + model.data.V);
226
227 return H;
228 }

239 template <typename Scalar, int nq, int ni>
240 Eigen::Matrix<Scalar, nq + nq, 1> forward_dynamics(Model<Scalar, nq>& model,
241 const Eigen::Matrix<Scalar, nq, 1>& q,
242 const Eigen::Matrix<Scalar, nq, 1>& p,
243 const Eigen::Matrix<Scalar, ni, 1>& u,
244 const std::vector<std::string>& active_constraints = {}) {
245 // If there are no active constraints, use the quicker forward dynamics
246 if (active_constraints.size() == 0) {
247 return forward_dynamics_without_constraints(model, q, p, u);
248 }
249 // Cast to autodiff type for automatic differentiation
250 Eigen::Matrix<autodiff::real, nq, 1> q_ad(q);
251 Eigen::Matrix<autodiff::real, nq, 1> p_ad(p);
252 auto model_ad = model.template cast<autodiff::real>();
253
254 // Compute the jacobian of the hamiltonian wrt q and p
255 Eigen::Matrix<autodiff::real, 1, 1> H_ad;
256 Eigen::Matrix<Scalar, 1, nq> dH_dq = autodiff::jacobian(hamiltonian_with_constraints<autodiff::real, nq>,
257 wrt(q_ad),
258 at(model_ad, q_ad, p_ad, active_constraints),
259 H_ad);
260
261 Eigen::Matrix<autodiff::real, Eigen::Dynamic, 1> pr_ad = p_ad.tail(model_ad.data.nr);
262 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> dH_dp = (model_ad.data.Mrinv * pr_ad).template cast<Scalar>();
263
264 Eigen::Matrix<Scalar, nq + nq, 1> dx_dt = Eigen::Matrix<Scalar, nq + nq, 1>::Zero();
265 // qdot
266 dx_dt.block(0, 0, nq, 1) = (model_ad.data.Jcp * dH_dp).template cast<Scalar>();
267
268 // pdot
269 dx_dt.block(nq + model_ad.data.nz, 0, model_ad.data.nr, 1) =
270 (-model_ad.data.Jcp.transpose() * dH_dq.transpose() - model_ad.data.Dr * dH_dp + model_ad.data.Gr * u)
271 .template cast<Scalar>();
272
273 // Store the result
274 model.data.dx_dt = dx_dt;
275 return dx_dt;
276 }

287 template <typename Scalar, int nq>
288 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> holonomic_constraints(
289 Model<Scalar, nq>& model,
290 const Eigen::Matrix<Scalar, nq, 1>& q,
291 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& M,
292 Scalar& V) {
293
294 std::vector<Scalar> Mp;
295 std::vector<Scalar> Jp;
296 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> fc;
297
298 // // Get the base link from the model
299 auto base_link = model.links[model.base_link_idx];
300
301 // Create over parametrised system
302 auto start = std::chrono::high_resolution_clock::now();
303 for (auto link : model.links) {
304 // Compute FK to centre of mass
305 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbm =
306 forward_kinematics_com<Scalar, nq>(model, q, base_link.name, link.name);
307 Eigen::Matrix<Scalar, 3, 1> rMBb = Hbm.translation();
308 // Add links contribution to potential energy m* g* h
309 V -= link.mass * model.gravity.transpose() * rMBb;
310
311 if (link.idx != -1 && link.joint.type == JointType::REVOLUTE) {
312 // Add to mass matrix list
313 Mp.insert(Mp.end(), {link.mass, link.mass, link.mass});
314 // Add inertia to J matrix TODO: Need to figure out inertia contribution
315 Jp.insert(Jp.end(), {0});
316 // Add to constraint vector
317 fc.conservativeResize(fc.rows() + 3);
318 fc.tail(3) = rMBb;
319 }
320 else if (link.idx != -1 && link.joint.type == JointType::FIXED) {
321 // Add to mass matrix list
322 Mp.insert(Mp.end(), {link.mass, link.mass, link.mass});
323 // Add to constraint vector
324 fc.conservativeResize(fc.rows() + 3);
325 fc.tail(3) = rMBb;
326 }
327 else if (link.idx != -1 && link.joint.type == JointType::PRISMATIC) {
328 // Add inertia to J matrix TODO: Need to figure out inertia contribution
329 Jp.insert(Jp.end(), {0});
330 }
331 }
332 // Compute reduced system
333 M.resize(Mp.size() + Jp.size(), Mp.size() + Jp.size());
334 M.setZero();
335 auto stop = std::chrono::high_resolution_clock::now();
336 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(stop - start);
337 // std::cout << "Builindg M " << duration.count() << " microseconds" << std::endl;
338 for (int i = 0; i < Jp.size(); i++) {
339 M(i, i) = Jp[i];
340 }
341 for (int i = 0; i < Mp.size(); i++) {
342 M(Jp.size() + i, Jp.size() + i) = Mp[i];
343 }
344 return fc;

356 template <typename Scalar, int nq>
357 void holonomic_reduction(Model<Scalar, nq>& model,
358 Eigen::Matrix<Scalar, nq, 1>& q_real,
359 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& M,
360 const Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& dfcdqh) {
361 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Mh =
362 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>::Zero(nq, nq);
363 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Q;
364 Q.resize(nq + dfcdqh.rows(), dfcdqh.cols());
365 Q.block(0, 0, nq, dfcdqh.cols()) = Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>::Identity(nq, nq);
366 Q.block(nq, 0, dfcdqh.rows(), dfcdqh.cols()) = dfcdqh.template cast<Scalar>();
367 Mh = Q.transpose() * M * Q;
368 M.resize(nq, nq);
369 M = Mh;

378 void compute_dynamics(Model<Scalar, nq>& model,
379 const Eigen::Matrix<Scalar, nq, 1>& q,
380 Eigen::Matrix<Scalar, nq, nq>& Mh,
381 Eigen::Matrix<Scalar, nq, nq>& Ch,
382 Eigen::Matrix<Scalar, nq, 1>& g,
383 Scalar& Vh) {
384 // Create the mass matrix, inertia matrix, constraint vector and potential energy matrices

410 template <typename Scalar, int nq>
411 std::vector<Eigen::Matrix<Scalar, 6, 1>> apply_external_forces(
412 const Model<Scalar, nq>& model,
413 const std::vector<Eigen::Matrix<Scalar, 6, 6>>& Xup,
414 const std::vector<Eigen::Matrix<Scalar, 6, 1>>& f_in,
415 const std::vector<Eigen::Matrix<Scalar, 6, 1>>& f_ext) {
416 std::vector<Eigen::Matrix<Scalar, 6, 1>> f_out(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
417 std::vector<Eigen::Matrix<Scalar, 6, 6>> Xa(nq, Eigen::Matrix<Scalar, 6, 6>::Zero());
418
419 f_out = f_in;
420 for (int i = 0; i < nq; i++) {
421 const auto link = model.links[model.q_idx[i]];
422 if (model.parent[i] == -1) {
423 Xa[i] = Xup[i];
424 }
425 else {
426 Xa[i] = Xup[i] * Xa[model.parent[i]];
427 }
428 f_out[i] += Xa[i].transpose().inverse() * f_ext[i];
429 }
430 return f_out;
431 }

443 template <typename Scalar, int nq>
444 Eigen::Matrix<Scalar, nq, 1> forward_dynamics_ab(const Model<Scalar, nq>& model,
445 const Eigen::Matrix<Scalar, nq, 1>& q,
446 const Eigen::Matrix<Scalar, nq, 1>& qd,
447 const Eigen::Matrix<Scalar, nq, 1>& tau,
448 const std::vector<Eigen::Matrix<Scalar, 6, 1>>& f_ext = {}) {
449 // Initialize the vectors for the algorithm
450 Eigen::Matrix<Scalar, 6, 1> g = Eigen::Matrix<Scalar, 6, 1>::Zero();
451 g.tail(3) = model.gravity;
452 std::vector<Eigen::Matrix<Scalar, 6, 6>> Xup(nq, Eigen::Matrix<Scalar, 6, 6>::Zero());
453 std::vector<Eigen::Matrix<Scalar, 6, 1>> S(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
454 std::vector<Eigen::Matrix<Scalar, 6, 1>> v(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
455 std::vector<Eigen::Matrix<Scalar, 6, 1>> c(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
456 std::vector<Eigen::Matrix<Scalar, 6, 6>> IA(nq, Eigen::Matrix<Scalar, 6, 6>::Zero());
457 std::vector<Eigen::Matrix<Scalar, 6, 1>> pA(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
458 std::vector<Eigen::Matrix<Scalar, 6, 1>> U(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
459 std::vector<Scalar> d(nq, 0);
460 std::vector<Scalar> u(nq, 0);
461 std::vector<Eigen::Matrix<Scalar, 6, 1>> a(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
462 Eigen::Matrix<Scalar, nq, 1> qdd = Eigen::Matrix<Scalar, nq, 1>::Zero();
463
464 for (int i = 0; i < nq; i++) {
465 // Compute the joint transform and motion subspace matrices
466 auto link = model.links[model.q_idx[i]];
467 S[i] = link.joint.S;
468 Eigen::Matrix<Scalar, 6, 1> vJ = S[i] * qd(i);
469 // Get transform from body to parent
470 auto T = link.joint.get_parent_to_child_transform(q(i));
471 // Compute the spatial transform from the parent to the current body
472 Xup[i] = homogeneous_to_spatial(T.inverse());
473 // Check if the model.parent link is the base link
474 if (model.parent[i] == -1) {
475 v[i] = vJ;
476 c[i] = Eigen::Matrix<Scalar, 6, 1>::Zero();
477 }
478 else {
479 v[i] = Xup[i] * v[model.parent[i]] + vJ;
480 c[i] = cross_spatial(v[i]) * vJ;
481 }
482 IA[i] = link.I;
483 pA[i] = cross_motion(v[i]) * IA[i] * v[i];
484 }
485
486 // Apply external forces if non-zero
487 if (f_ext.size() != 0) {
488 pA = apply_external_forces(model, Xup, pA, f_ext);
489 }
490
491 for (int i = nq - 1; i >= 0; i--) {
492 U[i] = IA[i] * S[i];
493 d[i] = S[i].transpose() * U[i];
494 u[i] = Scalar(tau(i) - S[i].transpose() * pA[i]);
495 if (model.parent[i] != -1) {
496 Eigen::Matrix<Scalar, 6, 6> Ia = IA[i] - (U[i] / d[i]) * U[i].transpose();
497 Eigen::Matrix<Scalar, 6, 1> pa = pA[i] + Ia * c[i] + U[i] * (u[i] / d[i]);
498 IA[model.parent[i]] += Xup[i].transpose() * Ia * Xup[i];
499 pA[model.parent[i]] += Xup[i].transpose() * pa;
500 }
501 }
502
503 for (int i = 0; i < nq; i++) {
504 if (model.parent[i] == -1) {
505 a[i] = Xup[i] * -g + c[i];
506 }
507 else {
508 a[i] = Xup[i] * a[model.parent[i]] + c[i];
509 }
510 qdd(i) = (u[i] - U[i].transpose() * a[i]) / d[i];
511 a[i] = a[i] + S[i] * qdd(i);
512 }
513 return qdd;
514 }
