27 template <typename Scalar, int nq>
28 Eigen::Matrix<Scalar, nq, nq> mass_matrix(Model<Scalar, nq>& model, const Eigen::Matrix<Scalar, nq, 1>& q) {
29 // Reset the mass matrix and potential energy
30 model.data.M.setZero();
31 model.data.V = 0;
32 Eigen::Matrix<Scalar, 6, 6> Mi;
33 // Get the base link from the model
34 auto base_link = model.links[model.base_link_idx];
35 for (int i = 0; i < model.n_links; i++) {
36 Mi.setZero();
37 // Compute the rotation between the ith link and the base
38 Eigen::Matrix<Scalar, 3, 3> R0i = rotation(model, q, base_link.name, model.links[i].name);
39 // Insert the mass of the link into the top 3 diagonals
40 Mi.block(0, 0, 3, 3) = model.links[i].mass * Eigen::Matrix<Scalar, 3, 3>::Identity();
41 // Insert the inertia of the link into the bottom 3 diagonals
42 Mi.block(3, 3, 3, 3) = R0i * model.links[i].inertia * R0i.transpose();
43 // Compute the geometric jacobian of the links center of mass with respect to the base
44 Eigen::Matrix<Scalar, 6, nq> Jci = geometric_jacobian_com(model, q, model.links[i].name);
45 // Compute the contribution to the mass matrix of the link
46 model.data.M += Jci.transpose() * Mi * Jci;
47 // Compute the contribution to the potential energy of the link
48 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbi_c =
49 forward_kinematics_com<Scalar, nq>(model, q, model.base_link_idx, model.links[i].idx);
50 Eigen::Matrix<Scalar, 3, 1> rMIi_c = Hbi_c.translation();
51 model.data.V += -model.links[i].mass * model.gravity.transpose() * rMIi_c;
52 }
53 return model.data.M;
54 }

98 template <typename Scalar, int nq>
99 Eigen::Matrix<Scalar, 1, 1> hamiltonian(Model<Scalar, nq>& model,
100 const Eigen::Matrix<Scalar, nq, 1>& q,
101 const Eigen::Matrix<Scalar, nq, 1>& p) {
102 // Compute the mass matrix and potential energy
103 mass_matrix<Scalar, nq>(model, q);
104
105 // Compute the inverse of the mass matrix
106 Eigen::Matrix<Scalar, nq, nq> b = Eigen::Matrix<Scalar, nq, nq>::Identity();
107 Eigen::Matrix<Scalar, nq, nq> Minv = model.data.M.ldlt().solve(b);
108 model.data.Minv = Minv;
109
110 // Compute the kinetic energy
111 model.data.T = Scalar(0.5 * p.transpose() * Minv * p);
112
113 // Compute the total energy
114 Eigen::Matrix<Scalar, 1, 1> H = Eigen::Matrix<Scalar, 1, 1>(model.data.T + model.data.V);
115 return H;
116 }

126 template <typename Scalar, int nq, int ni>
127 Eigen::Matrix<Scalar, nq + nq, 1> forward_dynamics_without_constraints(Model<Scalar, nq>& model,
128 const Eigen::Matrix<Scalar, nq, 1>& q,
129 const Eigen::Matrix<Scalar, nq, 1>& p,
130 const Eigen::Matrix<Scalar, ni, 1>& u) {
131 // Number of states
132 const int nx = nq + nq;
133 // Cast to autodiff type for automatic differentiation
134 Eigen::Matrix<autodiff::real, nq, 1> q_ad(q); // the input vector q
135 Eigen::Matrix<autodiff::real, nq, 1> p_ad(p); // the input vector p
136 auto model_ad = model.template cast<autodiff::real>();
137
138 // Compute the jacobian of the hamiltonian wrt q and p
139 Eigen::Matrix<autodiff::real, 1, 1> H_ad;
140
141
142 Eigen::Matrix<Scalar, 1, nq> dH_dq =
143 autodiff::jacobian(hamiltonian<autodiff::real, nq>, wrt(q_ad), at(model_ad, q_ad, p_ad), H_ad);
144
145 Eigen::Matrix<Scalar, 1, nq> dH_dp = (model_ad.data.Minv * p_ad).template cast<Scalar>();
146
147 // Create the interconnection and damping matrix
148 Eigen::Matrix<Scalar, nx, nx> J = Eigen::Matrix<Scalar, nx, nx>::Zero();
149 J.block(0, nq, nq, nq) = Eigen::Matrix<Scalar, nq, nq>::Identity();
150 J.block(nq, 0, nq, nq) = -1 * Eigen::Matrix<Scalar, nq, nq>::Identity();
151 J.block(nq, nq, nq, nq) = model.data.Dp;
152
153 // Stack the jacobians of the hamiltonian
154 Eigen::Matrix<Scalar, nx, 1> dH = Eigen::Matrix<Scalar, nx, 1>::Zero();
155 dH.block(0, 0, nq, 1) = dH_dq.transpose();
156 dH.block(nq, 0, nq, 1) = dH_dp.transpose();
157
158 // Compute the forward dynamics: dx_dt
159 Eigen::Matrix<Scalar, nx, ni> G = Eigen::Matrix<Scalar, nx, ni>::Zero();
160 G.block(nq, 0, nq, nq) = model.data.Gp;
161 Eigen::Matrix<Scalar, nx, 1> dx_dt = J * dH + G * u;
162
163 // Store the result
164 model.data.dx_dt = dx_dt;
165 return dx_dt;
166 }

175 template <typename Scalar, int nq>
176 Eigen::Matrix<Scalar, 1, 1> hamiltonian_with_constraints(Model<Scalar, nq>& model,
177 const Eigen::Matrix<Scalar, nq, 1>& q,
178 const Eigen::Matrix<Scalar, nq, 1>& p,
179 const std::vector<std::string>& active_constraints) {
180 // Compute the mass matrix and potential energy for unconstrained dynamics
181 mass_matrix<Scalar, nq>(model, q);
182
183 // Compute the jacobian of the active constraints
184 Eigen::Matrix<Scalar, Eigen::Dynamic, nq> Jc;
185 for (int i = 0; i < active_constraints.size(); i++) {
186 // Compute the jacobian of the active constraint
187 auto Jci = geometric_jacobian(model, q, active_constraints[i]);
188 auto Jci_v = Jci.block(0, 0, 3, nq);
189 // Vertically Concatenate the jacobian of the active constraint to the jacobian of the active constraints
190 Jc.conservativeResize(Jc.rows() + Jci_v.rows(), nq);
191 Jc.block(Jc.rows() - Jci_v.rows(), 0, Jci_v.rows(), Jci_v.cols()) = Jci_v;
192 }
193 model.data.Jc.resize(Jc.rows(), Jc.cols());
194 model.data.Jc = Jc;
195
196 // Compute the left annihilator of the jacobian of the active constraints
197 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Jcp = null<Scalar>(Jc);
198 model.data.Jcp.resize(Jcp.rows(), Jcp.cols());
199 model.data.Jcp = Jcp;
200
201 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Gr = Jcp.transpose() * model.data.Gp;
202 model.data.Gr.resize(Gr.rows(), Gr.cols());
203 model.data.Gr = Gr;
204
205 model.data.Dp.setZero();
206 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Dr = Jcp.transpose() * model.data.Dp * Jcp;
207 model.data.Dr.resize(Dr.rows(), Dr.cols());
208 model.data.Dr = Dr;
209
210 auto Mr = Jcp.transpose() * model.data.M * Jcp;
211 model.data.Mr.resize(Mr.rows(), Mr.cols());
212 model.data.Mr = Mr;
213
214 model.data.nz = model.data.Gp.rows() - Gr.rows();
215 model.data.nr = Mr.rows();
216
217 // Compute the inverse of the mass matrix
218 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> b;
219 b.resize(model.data.nr, model.data.nr);
220 b.setIdentity();
221 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Mrinv = Mr.ldlt().solve(b);
222 model.data.Mrinv.resize(model.data.nr, model.data.nr);
223 model.data.Mrinv = Mrinv;
224
225 // Compute the kinetic energy
226 auto pr = p.tail(model.data.nr);
227 model.data.T = Scalar(0.5 * pr.transpose() * Mrinv * pr);
228
229 // Compute the total energy
230 Eigen::Matrix<Scalar, 1, 1> H = Eigen::Matrix<Scalar, 1, 1>(model.data.T + model.data.V);
231
232 return H;
233 }

244 template <typename Scalar, int nq, int ni>
245 Eigen::Matrix<Scalar, nq + nq, 1> forward_dynamics(Model<Scalar, nq>& model,
246 const Eigen::Matrix<Scalar, nq, 1>& q,
247 const Eigen::Matrix<Scalar, nq, 1>& p,
248 const Eigen::Matrix<Scalar, ni, 1>& u,
249 const std::vector<std::string>& active_constraints = {}) {
250 // If there are no active constraints, use the quicker forward dynamics
251 if (active_constraints.size() == 0) {
252 return forward_dynamics_without_constraints(model, q, p, u);
253 }
254 // Cast to autodiff type for automatic differentiation
255 Eigen::Matrix<autodiff::real, nq, 1> q_ad(q);
256 Eigen::Matrix<autodiff::real, nq, 1> p_ad(p);
257 auto model_ad = model.template cast<autodiff::real>();
258
259 // Compute the jacobian of the hamiltonian wrt q and p
260 Eigen::Matrix<autodiff::real, 1, 1> H_ad;
261 Eigen::Matrix<Scalar, 1, nq> dH_dq = autodiff::jacobian(hamiltonian_with_constraints<autodiff::real, nq>,
262 wrt(q_ad),
263 at(model_ad, q_ad, p_ad, active_constraints),
264 H_ad);
265
266 Eigen::Matrix<autodiff::real, Eigen::Dynamic, 1> pr_ad = p_ad.tail(model_ad.data.nr);
267 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> dH_dp = (model_ad.data.Mrinv * pr_ad).template cast<Scalar>();
268
269 Eigen::Matrix<Scalar, nq + nq, 1> dx_dt = Eigen::Matrix<Scalar, nq + nq, 1>::Zero();
270 // qdot
271 dx_dt.block(0, 0, nq, 1) = (model_ad.data.Jcp * dH_dp).template cast<Scalar>();
272
273 // pdot
274 dx_dt.block(nq + model_ad.data.nz, 0, model_ad.data.nr, 1) =
275 (-model_ad.data.Jcp.transpose() * dH_dq.transpose() - model_ad.data.Dr * dH_dp + model_ad.data.Gr * u)
276 .template cast<Scalar>();
277
278 // Store the result
279 model.data.dx_dt = dx_dt;
280 return dx_dt;
281 }

292 template <typename Scalar, int nq>
293 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> holonomic_constraints(
294 Model<Scalar, nq>& model,
295 const Eigen::Matrix<Scalar, nq, 1>& q,
296 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& M,
297 Scalar& V) {
298
299 std::vector<Scalar> Mp;
300 std::vector<Scalar> Jp;
301 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> fc;
302
303 // // Get the base link from the model
304 auto base_link = model.links[model.base_link_idx];
305
306 // Create over parametrised system
307 auto start = std::chrono::high_resolution_clock::now();
308 for (auto link : model.links) {
309 // Compute FK to centre of mass
310 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbm =
311 forward_kinematics_com<Scalar, nq>(model, q, base_link.name, link.name);
312 Eigen::Matrix<Scalar, 3, 1> rMBb = Hbm.translation();
313 // Add links contribution to potential energy m* g* h
314 V -= link.mass * model.gravity.transpose() * rMBb;
315
316 if (link.idx != -1 && link.joint.type == JointType::REVOLUTE) {
317 // Add to mass matrix list
318 Mp.insert(Mp.end(), {link.mass, link.mass, link.mass});
319 // Add inertia to J matrix TODO: Need to figure out inertia contribution
320 Jp.insert(Jp.end(), {0});
321 // Add to constraint vector
322 fc.conservativeResize(fc.rows() + 3);
323 fc.tail(3) = rMBb;
324 }
325 else if (link.idx != -1 && link.joint.type == JointType::FIXED) {
326 // Add to mass matrix list
327 Mp.insert(Mp.end(), {link.mass, link.mass, link.mass});
328 // Add to constraint vector
329 fc.conservativeResize(fc.rows() + 3);
330 fc.tail(3) = rMBb;
331 }
332 else if (link.idx != -1 && link.joint.type == JointType::PRISMATIC) {
333 // Add inertia to J matrix TODO: Need to figure out inertia contribution
334 Jp.insert(Jp.end(), {0});
335 }
336 }
337 // Compute reduced system
338 M.resize(Mp.size() + Jp.size(), Mp.size() + Jp.size());
339 M.setZero();
340 auto stop = std::chrono::high_resolution_clock::now();
341 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(stop - start);
342 // std::cout << "Builindg M " << duration.count() << " microseconds" << std::endl;
343 for (int i = 0; i < Jp.size(); i++) {
344 M(i, i) = Jp[i];
345 }
346 for (int i = 0; i < Mp.size(); i++) {
347 M(Jp.size() + i, Jp.size() + i) = Mp[i];
348 }
349 return fc;
350 }

361 template <typename Scalar, int nq>
362 void holonomic_reduction(Model<Scalar, nq>& model,
363 Eigen::Matrix<Scalar, nq, 1>& q_real,
364 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& M,
365 const Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& dfcdqh) {
366 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Mh =
367 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>::Zero(nq, nq);
368 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Q;
369 Q.resize(nq + dfcdqh.rows(), dfcdqh.cols());
370 Q.block(0, 0, nq, dfcdqh.cols()) = Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>::Identity(nq, nq);
371 Q.block(nq, 0, dfcdqh.rows(), dfcdqh.cols()) = dfcdqh.template cast<Scalar>();
372 Mh = Q.transpose() * M * Q;
373 M.resize(nq, nq);
374 M = Mh;
375 }

382 template <typename Scalar, int nq>
383 void compute_dynamics(Model<Scalar, nq>& model,
384 const Eigen::Matrix<Scalar, nq, 1>& q,
385 Eigen::Matrix<Scalar, nq, nq>& Mh,
386 Eigen::Matrix<Scalar, nq, nq>& Ch,
387 Eigen::Matrix<Scalar, nq, 1>& g,
388 Scalar& Vh) {
389 // Create the mass matrix, inertia matrix, constraint vector and potential energy matrices
390 Eigen::Matrix<autodiff::real, Eigen::Dynamic, Eigen::Dynamic> M;
391 Eigen::Matrix<autodiff::real, Eigen::Dynamic, 1> fc;
392 autodiff::real V = 0;
393
394 // Cast to autodiff::real type
395 Eigen::Matrix<autodiff::real, nq, 1> q_real(q);
396 auto autodiff_model = model.template cast<autodiff::real>();
397
398 // Compute the holonomic constraint vector and its jacobian
399 Eigen::Matrix<autodiff::real, Eigen::Dynamic, 1> F;
400 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> dfcdqh =
401 jacobian(holonomic_constraints<autodiff::real, nq>, wrt(q_real), at(autodiff_model, q_real, M, V), F);
402
403 // Compute the mass matrix via holonomic constraint elimination
404 holonomic_reduction<autodiff::real, nq>(autodiff_model, q_real, M, dfcdqh);
405 };

415 template <typename Scalar, int nq>
416 std::vector<Eigen::Matrix<Scalar, 6, 1>> apply_external_forces(
417 const Model<Scalar, nq>& model,
418 const std::vector<Eigen::Matrix<Scalar, 6, 6>>& Xup,
419 const std::vector<Eigen::Matrix<Scalar, 6, 1>>& f_in,
420 const std::vector<Eigen::Matrix<Scalar, 6, 1>>& f_ext) {
421 std::vector<Eigen::Matrix<Scalar, 6, 1>> f_out(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
422 std::vector<Eigen::Matrix<Scalar, 6, 6>> Xa(nq, Eigen::Matrix<Scalar, 6, 6>::Zero());
423
424 f_out = f_in;
425 for (int i = 0; i < nq; i++) {
426 const auto link = model.links[model.q_idx[i]];
427 if (model.parent[i] == -1) {
428 Xa[i] = Xup[i];
429 }
430 else {
431 Xa[i] = Xup[i] * Xa[model.parent[i]];
432 }
433 f_out[i] += Xa[i].transpose().inverse() * f_ext[i];
434 }
435 return f_out;
436 }

448 template <typename Scalar, int nq>
449 Eigen::Matrix<Scalar, nq, 1> forward_dynamics_ab(const Model<Scalar, nq>& model,
450 const Eigen::Matrix<Scalar, nq, 1>& q,
451 const Eigen::Matrix<Scalar, nq, 1>& qd,
452 const Eigen::Matrix<Scalar, nq, 1>& tau,
453 const std::vector<Eigen::Matrix<Scalar, 6, 1>>& f_ext = {}) {
454 // Initialize the vectors for the algorithm
455 Eigen::Matrix<Scalar, 6, 1> g = Eigen::Matrix<Scalar, 6, 1>::Zero();
456 g.tail(3) = model.gravity;
457 std::vector<Eigen::Matrix<Scalar, 6, 6>> Xup(nq, Eigen::Matrix<Scalar, 6, 6>::Zero());
458 std::vector<Eigen::Matrix<Scalar, 6, 1>> S(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
459 std::vector<Eigen::Matrix<Scalar, 6, 1>> v(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
460 std::vector<Eigen::Matrix<Scalar, 6, 1>> c(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
461 std::vector<Eigen::Matrix<Scalar, 6, 6>> IA(nq, Eigen::Matrix<Scalar, 6, 6>::Zero());
462 std::vector<Eigen::Matrix<Scalar, 6, 1>> pA(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
463 std::vector<Eigen::Matrix<Scalar, 6, 1>> U(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
464 std::vector<Scalar> d(nq, 0);
465 std::vector<Scalar> u(nq, 0);
466 std::vector<Eigen::Matrix<Scalar, 6, 1>> a(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
467 Eigen::Matrix<Scalar, nq, 1> qdd = Eigen::Matrix<Scalar, nq, 1>::Zero();
468
469 for (int i = 0; i < nq; i++) {
470 // Compute the joint transform and motion subspace matrices
471 auto link = model.links[model.q_idx[i]];
472 S[i] = link.joint.S;
473 Eigen::Matrix<Scalar, 6, 1> vJ = S[i] * qd(i);
474 // Get transform from body to parent
475 auto T = link.joint.get_parent_to_child_transform(q(i));
476 // Compute the spatial transform from the parent to the current body
477 Xup[i] = homogeneous_to_spatial(T.inverse());
478 // Check if the model.parent link is the base link
479 if (model.parent[i] == -1) {
480 v[i] = vJ;
481 c[i] = Eigen::Matrix<Scalar, 6, 1>::Zero();
482 }
483 else {
484 v[i] = Xup[i] * v[model.parent[i]] + vJ;
485 c[i] = cross_spatial(v[i]) * vJ;
486 }
487 IA[i] = link.I;
488 pA[i] = cross_motion(v[i]) * IA[i] * v[i];
489 }
490
491 // Apply external forces if non-zero
492 if (f_ext.size() != 0) {
493 pA = apply_external_forces(model, Xup, pA, f_ext);
494 }
495
496 for (int i = nq - 1; i >= 0; i--) {
497 U[i] = IA[i] * S[i];
498 d[i] = S[i].transpose() * U[i];
499 u[i] = Scalar(tau(i) - S[i].transpose() * pA[i]);
500 if (model.parent[i] != -1) {
501 Eigen::Matrix<Scalar, 6, 6> Ia = IA[i] - (U[i] / d[i]) * U[i].transpose();
502 Eigen::Matrix<Scalar, 6, 1> pa = pA[i] + Ia * c[i] + U[i] * (u[i] / d[i]);
503 IA[model.parent[i]] += Xup[i].transpose() * Ia * Xup[i];
504 pA[model.parent[i]] += Xup[i].transpose() * pa;
505 }
506 }
507
508 for (int i = 0; i < nq; i++) {
509 if (model.parent[i] == -1) {
510 a[i] = Xup[i] * -g + c[i];
511 }
512 else {
513 a[i] = Xup[i] * a[model.parent[i]] + c[i];
514 }
515 qdd(i) = (u[i] - U[i].transpose() * a[i]) / d[i];
516 a[i] = a[i] + S[i] * qdd(i);
517 }
518 return qdd;
519 }
