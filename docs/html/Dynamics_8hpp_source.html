20 template <typename Scalar, int nq>
21 Eigen::Matrix<Scalar, nq, nq> mass_matrix(Model<Scalar, nq>& model, const Eigen::Matrix<Scalar, nq, 1>& q) {
22 // Reset the mass matrix and potential energy
23 model.data.M.setZero();
24 model.data.V = 0;
25 Eigen::Matrix<Scalar, 6, 6> Mi;
26 // Get the base link from the model
27 auto base_link = model.links[model.base_link_idx];
28 for (int i = 0; i < model.n_links; i++) {
29 Mi.setZero();
30 // Compute the rotation between the ith link and the base
31 Eigen::Matrix<Scalar, 3, 3> R0i = rotation(model, q, base_link.name, model.links[i].name);
32 // Insert the mass of the link into the top 3 diagonals
33 Mi.block(0, 0, 3, 3) = model.links[i].mass * Eigen::Matrix<Scalar, 3, 3>::Identity();
34 // Insert the inertia of the link into the bottom 3 diagonals
35 Mi.block(3, 3, 3, 3) = R0i * model.links[i].inertia * R0i.transpose();
36 // Compute the geometric jacobian of the links center of mass with respect to the base
37 Eigen::Matrix<Scalar, 6, nq> Jci = geometric_jacobian_com(model, q, model.links[i].name);
38 // Compute the contribution to the mass matrix of the link
39 model.data.M += Jci.transpose() * Mi * Jci;
40 // Compute the contribution to the potential energy of the link
41 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbi_c =
42 forward_kinematics_com<Scalar, nq>(model, q, model.base_link_idx, model.links[i].idx);
43 Eigen::Matrix<Scalar, 3, 1> rMIi_c = Hbi_c.translation();
44 model.data.V += -model.links[i].mass * model.gravity.transpose() * rMIi_c;
45 }
46 return model.data.M;
47 }

91 template <typename Scalar, int nq>
92 Eigen::Matrix<Scalar, 1, 1> hamiltonian(Model<Scalar, nq>& model,
93 const Eigen::Matrix<Scalar, nq, 1>& q,
94 const Eigen::Matrix<Scalar, nq, 1>& p) {
95 // Compute the mass matrix and potential energy
96 mass_matrix<Scalar, nq>(model, q);
97
98 // Compute the inverse of the mass matrix
99 Eigen::Matrix<Scalar, nq, nq> b = Eigen::Matrix<Scalar, nq, nq>::Identity();
100 Eigen::Matrix<Scalar, nq, nq> Minv = model.data.M.ldlt().solve(b);
101 model.data.Minv = Minv;
102
103 // Compute the kinetic energy
104 model.data.T = Scalar(0.5 * p.transpose() * Minv * p);
105
106 // Compute the total energy
107 Eigen::Matrix<Scalar, 1, 1> H = Eigen::Matrix<Scalar, 1, 1>(model.data.T + model.data.V);
108 return H;
109 }

119 template <typename Scalar, int nq, int ni>
120 Eigen::Matrix<Scalar, nq + nq, 1> forward_dynamics_without_constraints(Model<Scalar, nq>& model,
121 const Eigen::Matrix<Scalar, nq, 1>& q,
122 const Eigen::Matrix<Scalar, nq, 1>& p,
123 const Eigen::Matrix<Scalar, ni, 1>& u) {
124 // Number of states
125 const int nx = nq + nq;
126 // Cast to autodiff type for automatic differentiation
127 Eigen::Matrix<autodiff::real, nq, 1> q_ad(q); // the input vector q
128 Eigen::Matrix<autodiff::real, nq, 1> p_ad(p); // the input vector p
129 auto model_ad = model.template cast<autodiff::real>();
130
131 // Compute the jacobian of the hamiltonian wrt q and p
132 Eigen::Matrix<autodiff::real, 1, 1> H_ad;
133
134
135 Eigen::Matrix<Scalar, 1, nq> dH_dq =
136 autodiff::jacobian(hamiltonian<autodiff::real, nq>, wrt(q_ad), at(model_ad, q_ad, p_ad), H_ad);
137
138 Eigen::Matrix<Scalar, 1, nq> dH_dp = (model_ad.data.Minv * p_ad).template cast<Scalar>();
139
140 // Create the interconnection and damping matrix
141 Eigen::Matrix<Scalar, nx, nx> J = Eigen::Matrix<Scalar, nx, nx>::Zero();
142 J.block(0, nq, nq, nq) = Eigen::Matrix<Scalar, nq, nq>::Identity();
143 J.block(nq, 0, nq, nq) = -1 * Eigen::Matrix<Scalar, nq, nq>::Identity();
144 J.block(nq, nq, nq, nq) = model.data.Dp;
145
146 // Stack the jacobians of the hamiltonian
147 Eigen::Matrix<Scalar, nx, 1> dH = Eigen::Matrix<Scalar, nx, 1>::Zero();
148 dH.block(0, 0, nq, 1) = dH_dq.transpose();
149 dH.block(nq, 0, nq, 1) = dH_dp.transpose();
150
151 // Compute the forward dynamics: dx_dt
152 Eigen::Matrix<Scalar, nx, ni> G = Eigen::Matrix<Scalar, nx, ni>::Zero();
153 G.block(nq, 0, nq, nq) = model.data.Gp;
154 Eigen::Matrix<Scalar, nx, 1> dx_dt = J * dH + G * u;
155
156 // Store the result
157 model.data.dx_dt = dx_dt;
158 return dx_dt;
159 }

168 template <typename Scalar, int nq>
169 Eigen::Matrix<Scalar, 1, 1> hamiltonian_with_constraints(Model<Scalar, nq>& model,
170 const Eigen::Matrix<Scalar, nq, 1>& q,
171 const Eigen::Matrix<Scalar, nq, 1>& p,
172 const std::vector<std::string>& active_constraints) {
173 // Compute the mass matrix and potential energy for unconstrained dynamics
174 mass_matrix<Scalar, nq>(model, q);
175
176 // Compute the jacobian of the active constraints
177 Eigen::Matrix<Scalar, Eigen::Dynamic, nq> Jc;
178 for (int i = 0; i < active_constraints.size(); i++) {
179 // Compute the jacobian of the active constraint
180 auto Jci = tr::geometric_jacobian(model, q, active_constraints[i]);
181 auto Jci_v = Jci.block(0, 0, 3, nq);
182 // Vertically Concatenate the jacobian of the active constraint to the jacobian of the active constraints
183 Jc.conservativeResize(Jc.rows() + Jci_v.rows(), nq);
184 Jc.block(Jc.rows() - Jci_v.rows(), 0, Jci_v.rows(), Jci_v.cols()) = Jci_v;
185 }
186 model.data.Jc.resize(Jc.rows(), Jc.cols());
187 model.data.Jc = Jc;
188
189 // Compute the left annihilator of the jacobian of the active constraints
190 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Jcp = tr::null<Scalar>(Jc);
191 model.data.Jcp.resize(Jcp.rows(), Jcp.cols());
192 model.data.Jcp = Jcp;
193
194 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Gr = Jcp.transpose() * model.data.Gp;
195 model.data.Gr.resize(Gr.rows(), Gr.cols());
196 model.data.Gr = Gr;
197
198 model.data.Dp.setZero();
199 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Dr = Jcp.transpose() * model.data.Dp * Jcp;
200 model.data.Dr.resize(Dr.rows(), Dr.cols());
201 model.data.Dr = Dr;
202
203 auto Mr = Jcp.transpose() * model.data.M * Jcp;
204 model.data.Mr.resize(Mr.rows(), Mr.cols());
205 model.data.Mr = Mr;
206
207 model.data.nz = model.data.Gp.rows() - Gr.rows();
208 model.data.nr = Mr.rows();
209
210 // Compute the inverse of the mass matrix
211 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> b;
212 b.resize(model.data.nr, model.data.nr);
213 b.setIdentity();
214 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Mrinv = Mr.ldlt().solve(b);
215 model.data.Mrinv.resize(model.data.nr, model.data.nr);
216 model.data.Mrinv = Mrinv;
217
218 // Compute the kinetic energy
219 auto pr = p.tail(model.data.nr);
220 model.data.T = Scalar(0.5 * pr.transpose() * Mrinv * pr);
221
222 // Compute the total energy
223 Eigen::Matrix<Scalar, 1, 1> H = Eigen::Matrix<Scalar, 1, 1>(model.data.T + model.data.V);
224
225 return H;
226 }

237 template <typename Scalar, int nq, int ni>
238 Eigen::Matrix<Scalar, nq + nq, 1> forward_dynamics(Model<Scalar, nq>& model,
239 const Eigen::Matrix<Scalar, nq, 1>& q,
240 const Eigen::Matrix<Scalar, nq, 1>& p,
241 const Eigen::Matrix<Scalar, ni, 1>& u,
242 const std::vector<std::string>& active_constraints = {}) {
243 // If there are no active constraints, use the quicker forward dynamics
244 if (active_constraints.size() == 0) {
245 return forward_dynamics_without_constraints(model, q, p, u);
246 }
247 // Cast to autodiff type for automatic differentiation
248 Eigen::Matrix<autodiff::real, nq, 1> q_ad(q);
249 Eigen::Matrix<autodiff::real, nq, 1> p_ad(p);
250 auto model_ad = model.template cast<autodiff::real>();
251
252 // Compute the jacobian of the hamiltonian wrt q and p
253 Eigen::Matrix<autodiff::real, 1, 1> H_ad;
254 Eigen::Matrix<Scalar, 1, nq> dH_dq = autodiff::jacobian(hamiltonian_with_constraints<autodiff::real, nq>,
255 wrt(q_ad),
256 at(model_ad, q_ad, p_ad, active_constraints),
257 H_ad);
258
259 Eigen::Matrix<autodiff::real, Eigen::Dynamic, 1> pr_ad = p_ad.tail(model_ad.data.nr);
260 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> dH_dp = (model_ad.data.Mrinv * pr_ad).template cast<Scalar>();
261
262 Eigen::Matrix<Scalar, nq + nq, 1> dx_dt = Eigen::Matrix<Scalar, nq + nq, 1>::Zero();
263 // qdot
264 dx_dt.block(0, 0, nq, 1) = (model_ad.data.Jcp * dH_dp).template cast<Scalar>();
265
266 // pdot
267 dx_dt.block(nq + model_ad.data.nz, 0, model_ad.data.nr, 1) =
268 (-model_ad.data.Jcp.transpose() * dH_dq.transpose() - model_ad.data.Dr * dH_dp + model_ad.data.Gr * u)
269 .template cast<Scalar>();
270
271 // Store the result
272 model.data.dx_dt = dx_dt;
273 return dx_dt;
274 }

285 template <typename Scalar, int nq>
286 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> holonomic_constraints(
287 Model<Scalar, nq>& model,
288 const Eigen::Matrix<Scalar, nq, 1>& q,
289 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& M,
290 Scalar& V) {
291
292 std::vector<Scalar> Mp;
293 std::vector<Scalar> Jp;
294 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> fc;
295
296 // // Get the base link from the model
297 auto base_link = model.links[model.base_link_idx];
298
299 // Create over parametrised system
300 auto start = std::chrono::high_resolution_clock::now();
301 for (auto link : model.links) {
302 // Compute FK to centre of mass
303 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbm =
304 forward_kinematics_com<Scalar, nq>(model, q, base_link.name, link.name);
305 Eigen::Matrix<Scalar, 3, 1> rMBb = Hbm.translation();
306 // Add links contribution to potential energy m* g* h
307 V -= link.mass * model.gravity.transpose() * rMBb;
308
309 if (link.idx != -1 && link.joint.type == tr::JointType::REVOLUTE) {
310 // Add to mass matrix list
311 Mp.insert(Mp.end(), {link.mass, link.mass, link.mass});
312 // Add inertia to J matrix TODO: Need to figure out inertia contribution
313 Jp.insert(Jp.end(), {0});
314 // Add to constraint vector
315 fc.conservativeResize(fc.rows() + 3);
316 fc.tail(3) = rMBb;
317 }
318 else if (link.idx != -1 && link.joint.type == tr::JointType::FIXED) {
319 // Add to mass matrix list
320 Mp.insert(Mp.end(), {link.mass, link.mass, link.mass});
321 // Add to constraint vector
322 fc.conservativeResize(fc.rows() + 3);
323 fc.tail(3) = rMBb;
324 }
325 else if (link.idx != -1 && link.joint.type == tr::JointType::PRISMATIC) {
326 // Add inertia to J matrix TODO: Need to figure out inertia contribution
327 Jp.insert(Jp.end(), {0});
328 }
329 }
330 // Compute reduced system
331 M.resize(Mp.size() + Jp.size(), Mp.size() + Jp.size());
332 M.setZero();
333 auto stop = std::chrono::high_resolution_clock::now();
334 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(stop - start);
335 // std::cout << "Builindg M " << duration.count() << " microseconds" << std::endl;
336 for (int i = 0; i < Jp.size(); i++) {
337 M(i, i) = Jp[i];
338 }
339 for (int i = 0; i < Mp.size(); i++) {
340 M(Jp.size() + i, Jp.size() + i) = Mp[i];
341 }
342 return fc;
343 }

354 template <typename Scalar, int nq>
355 void holonomic_reduction(Model<Scalar, nq>& model,
356 Eigen::Matrix<Scalar, nq, 1>& q_real,
357 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& M,
358 const Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& dfcdqh) {
359 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Mh =
360 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>::Zero(nq, nq);
361 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Q;
362 Q.resize(nq + dfcdqh.rows(), dfcdqh.cols());
363 Q.block(0, 0, nq, dfcdqh.cols()) = Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>::Identity(nq, nq);
364 Q.block(nq, 0, dfcdqh.rows(), dfcdqh.cols()) = dfcdqh.template cast<Scalar>();
365 Mh = Q.transpose() * M * Q;
366 M.resize(nq, nq);
367 M = Mh;
368 }

375 template <typename Scalar, int nq>
376 void compute_dynamics(Model<Scalar, nq>& model,
377 const Eigen::Matrix<Scalar, nq, 1>& q,
378 Eigen::Matrix<Scalar, nq, nq>& Mh,
379 Eigen::Matrix<Scalar, nq, nq>& Ch,
380 Eigen::Matrix<Scalar, nq, 1>& g,
381 Scalar& Vh) {
382 // Create the mass matrix, inertia matrix, constraint vector and potential energy matrices
383 Eigen::Matrix<autodiff::real, Eigen::Dynamic, Eigen::Dynamic> M;
384 Eigen::Matrix<autodiff::real, Eigen::Dynamic, 1> fc;
385 autodiff::real V = 0;
386
387 // Cast to autodiff::real type
388 Eigen::Matrix<autodiff::real, nq, 1> q_real(q);
389 auto autodiff_model = model.template cast<autodiff::real>();
390
391 // Compute the holonomic constraint vector and its jacobian
392 Eigen::Matrix<autodiff::real, Eigen::Dynamic, 1> F;
393 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> dfcdqh =
394 jacobian(holonomic_constraints<autodiff::real, nq>, wrt(q_real), at(autodiff_model, q_real, M, V), F);
395
396 // Compute the mass matrix via holonomic constraint elimination
397 holonomic_reduction<autodiff::real, nq>(autodiff_model, q_real, M, dfcdqh);
398 };

408 template <typename Scalar, int nq>
409 std::vector<Eigen::Matrix<Scalar, 6, 1>> apply_external_forces(
410 const tr::Model<Scalar, nq>& model,
411 const std::vector<Eigen::Matrix<Scalar, 6, 6>>& Xup,
412 const std::vector<Eigen::Matrix<Scalar, 6, 1>>& f_in,
413 const std::vector<Eigen::Matrix<Scalar, 6, 1>>& f_ext) {
414 std::vector<Eigen::Matrix<Scalar, 6, 1>> f_out(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
415 std::vector<Eigen::Matrix<Scalar, 6, 6>> Xa(nq, Eigen::Matrix<Scalar, 6, 6>::Zero());
416
417 f_out = f_in;
418 for (int i = 0; i < nq; i++) {
419 const auto link = model.links[model.q_idx[i]];
420 if (model.parent[i] == -1) {
421 Xa[i] = Xup[i];
422 }
423 else {
424 Xa[i] = Xup[i] * Xa[model.parent[i]];
425 }
426 f_out[i] += Xa[i].transpose().inverse() * f_ext[i];
427 }
428 return f_out;
429 }

441 template <typename Scalar, int nq>
442 Eigen::Matrix<Scalar, nq, 1> forward_dynamics_ab(const tr::Model<Scalar, nq>& model,
443 const Eigen::Matrix<Scalar, nq, 1>& q,
444 const Eigen::Matrix<Scalar, nq, 1>& qd,
445 const Eigen::Matrix<Scalar, nq, 1>& tau,
446 const std::vector<Eigen::Matrix<Scalar, 6, 1>>& f_ext = {}) {
447 // Initialize the vectors for the algorithm
448 Eigen::Matrix<Scalar, 6, 1> g = Eigen::Matrix<Scalar, 6, 1>::Zero();
449 g.tail(3) = model.gravity;
450 std::vector<Eigen::Matrix<Scalar, 6, 6>> Xup(nq, Eigen::Matrix<Scalar, 6, 6>::Zero());
451 std::vector<Eigen::Matrix<Scalar, 6, 1>> S(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
452 std::vector<Eigen::Matrix<Scalar, 6, 1>> v(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
453 std::vector<Eigen::Matrix<Scalar, 6, 1>> c(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
454 std::vector<Eigen::Matrix<Scalar, 6, 6>> IA(nq, Eigen::Matrix<Scalar, 6, 6>::Zero());
455 std::vector<Eigen::Matrix<Scalar, 6, 1>> pA(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
456 std::vector<Eigen::Matrix<Scalar, 6, 1>> U(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
457 std::vector<Scalar> d(nq, 0);
458 std::vector<Scalar> u(nq, 0);
459 std::vector<Eigen::Matrix<Scalar, 6, 1>> a(nq, Eigen::Matrix<Scalar, 6, 1>::Zero());
460 Eigen::Matrix<Scalar, nq, 1> qdd = Eigen::Matrix<Scalar, nq, 1>::Zero();
461
462 for (int i = 0; i < nq; i++) {
463 // Compute the joint transform and motion subspace matrices
464 auto link = model.links[model.q_idx[i]];
465 S[i] = link.joint.S;
466 Eigen::Matrix<Scalar, 6, 1> vJ = S[i] * qd(i);
467 // Get transform from body to parent
468 auto T = link.joint.get_parent_to_child_transform(q(i));
469 // Compute the spatial transform from the parent to the current body
470 Xup[i] = tr::homogeneous_to_spatial(T.inverse());
471 // Check if the model.parent link is the base link
472 if (model.parent[i] == -1) {
473 v[i] = vJ;
474 c[i] = Eigen::Matrix<Scalar, 6, 1>::Zero();
475 }
476 else {
477 v[i] = Xup[i] * v[model.parent[i]] + vJ;
478 c[i] = tr::cross_spatial(v[i]) * vJ;
479 }
480 IA[i] = link.I;
481 pA[i] = tr::cross_motion(v[i]) * IA[i] * v[i];
482 }
483
484 // Apply external forces if non-zero
485 if (f_ext.size() != 0) {
486 pA = apply_external_forces(model, Xup, pA, f_ext);
487 }
488
489 for (int i = nq - 1; i >= 0; i--) {
490 U[i] = IA[i] * S[i];
491 d[i] = S[i].transpose() * U[i];
492 u[i] = Scalar(tau(i) - S[i].transpose() * pA[i]);
493 if (model.parent[i] != -1) {
494 Eigen::Matrix<Scalar, 6, 6> Ia = IA[i] - (U[i] / d[i]) * U[i].transpose();
495 Eigen::Matrix<Scalar, 6, 1> pa = pA[i] + Ia * c[i] + U[i] * (u[i] / d[i]);
496 IA[model.parent[i]] += Xup[i].transpose() * Ia * Xup[i];
497 pA[model.parent[i]] += Xup[i].transpose() * pa;
498 }
499 }
500
501 for (int i = 0; i < nq; i++) {
502 if (model.parent[i] == -1) {
503 a[i] = Xup[i] * -g + c[i];
504 }
505 else {
506 a[i] = Xup[i] * a[model.parent[i]] + c[i];
507 }
508 qdd(i) = (u[i] - U[i].transpose() * a[i]) / d[i];
509 a[i] = a[i] + S[i] * qdd(i);
510 }
511 return qdd;
512 }
