22 template <typename Scalar, int nq>
23 Eigen::Transform<Scalar, 3, Eigen::Isometry> forward_kinematics(const Model<Scalar, nq>& model,
24 const Eigen::Matrix<Scalar, nq, 1>& q,
25 const int& target_link_idx,
26 const bool inverse_transform = false) {
27 // Get the target link
28 Link<Scalar> current_link = model.links[target_link_idx];
29
30 // Check if the link is within the kinematic tree
31 if (current_link.idx == -1) {
32 std::string error_msg = "Error: Link " + std::to_string(target_link_idx) + " not found.";
33 throw std::runtime_error(error_msg);
34 }
35
36 // Return identity transform if the link is itself the base link
37 if (current_link.idx == model.base_link_idx) {
38 return Eigen::Transform<Scalar, 3, Eigen::Isometry>::Identity();
39 }
40
41 // Build kinematic tree from target_link {t} to base {b}
42 Eigen::Transform<Scalar, 3, Eigen::Isometry> Htb = Eigen::Transform<Scalar, 3, Eigen::Isometry>::Identity();
43 while (current_link.idx != model.links[model.base_link_idx].idx) {
44 Eigen::Transform<Scalar, 3, Eigen::Isometry> H = Eigen::Transform<Scalar, 3, Eigen::Isometry>::Identity();
45 auto current_joint = current_link.joint;
46 if (current_joint.type == JointType::REVOLUTE) {
47 // Rotate by q_current around axis
48 H.linear() =
49 Eigen::AngleAxis<Scalar>(Scalar(q(current_joint.idx)), current_joint.axis).toRotationMatrix();
50 }
51 else if (current_joint.type == JointType::PRISMATIC) {
52 // Translate by q_current along axis
53 H.translation() = current_joint.axis * Scalar(q(current_joint.idx));
54 }
55 Htb = Htb * H.inverse();
56 // Apply inverse joint transform as we are going back up tree
57 Htb = Htb * current_joint.parent_transform.inverse();
58 // Move up the tree to parent link
59 current_link = model.links[current_link.parent];
60 }
61
62 // If inverse transform is requested, return Hbt, which is transform from target {t} to base {b}
63 if (inverse_transform) {
64 return Htb;
65 }
66 // Return transform Hbt, which is transform from base {b} to target {t}
67 return Htb.inverse();
68 }

100 template <typename Scalar, int nq>
101 Eigen::Transform<Scalar, 3, Eigen::Isometry> forward_kinematics(const Model<Scalar, nq>& model,
102 const Eigen::Matrix<Scalar, nq, 1>& q,
103 const std::string& source_link_name,
104 const std::string& target_link_name) {
105 // Build kinematic tree from source {s} to base {b} frame
106 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hsb = forward_kinematics(model, q, source_link_name, true);
107 // Build kinematic tree from base {b} to target {t} frame
108 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbt = forward_kinematics(model, q, target_link_name);
109 // Compute transform between source {s} and target {t} frames
110 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hst = Hsb * Hbt;
111 return Hst;
112 }

150 const Eigen::Matrix<Scalar, nq, 1>& q,
151 const std::string& source_link_name,
152 const std::string& target_link_name) {
153 // Compute forward kinematics from source {b} to target {t}

256 template <typename Scalar, int nq>
257 Eigen::Matrix<Scalar, 6, nq> geometric_jacobian(const Model<Scalar, nq>& model,
258 const Eigen::Matrix<Scalar, nq, 1>& q,
259 const int& target_link_idx) {
260 // Initialize the geometric jabobian matrix with zeros
261 Eigen::Matrix<Scalar, 6, nq> J = Eigen::Matrix<Scalar, 6, nq>::Zero();
262
263 // Get the target link from the model
264 Link<Scalar> current_link = model.links[target_link_idx];
265
266 // Get the base link from the model
267 auto base_link = model.links[model.base_link_idx];
268
269 // Compute the displacement of the target link {t} in the base link frame {b}
270 Eigen::Matrix<Scalar, 3, 1> rTBb = translation(model, q, base_link.idx, current_link.idx);
271
272 while (current_link.idx != model.base_link_idx) {
273 auto current_joint = current_link.joint;
274 if (current_joint.idx != -1) {
275 // Compute the transform between base {b} and the current link {i}
276 auto Hbi = forward_kinematics(model, q, base_link.idx, current_link.idx);
277 // Axis of the current joint rotated into the base frame {b}
278 auto zIBb = Hbi.linear() * current_joint.axis;
279 // Compute the displacement of the current link {i} from the base link frame {b}
280 auto rIBb = Hbi.translation();
281 if (current_joint.type == JointType::PRISMATIC) {
282 J.block(0, current_joint.idx, 3, 1) = zIBb;
283 J.block(3, current_joint.idx, 3, 1) = Eigen::Matrix<Scalar, 3, 1>::Zero();
284 }
285 else if (current_joint.type == JointType::REVOLUTE) {
286 J.block(0, current_joint.idx, 3, 1) = (zIBb).cross(rTBb - rIBb);
287 J.block(3, current_joint.idx, 3, 1) = zIBb;
288 }
289 }
290 // Move up the tree to parent towards the base
291 current_link = model.links[current_link.parent];
292 }
293 return J;
294 }

373 template <typename Scalar, int nq>
374 Eigen::Matrix<Scalar, 3, 1> centre_of_mass(const Model<Scalar, nq>& model,
375 const Eigen::Matrix<Scalar, nq, 1>& q,
376 const std::string& source_link_name) {
377 // For each link in the model, compute the transform from the source link to the CoM of the link
378 Scalar total_mass = 0;
379 Eigen::Matrix<Scalar, 3, 1> rISs = Eigen::Matrix<Scalar, 3, 1>::Zero();
380 for (auto link : model.links) {
381 // Compute the transform from the source link to the CoM of the link
382 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hsc =
383 forward_kinematics_com(model, q, source_link_name, link->name);
384 // Compute the centre of mass of the link
385 rISs = rISs + Hsc.translation() * link->mass;
386 // Add the links mass to the total mass
387 total_mass += link->mass;
388 }
389 // Compute the centre of mass from the source link
390 return rISs / total_mass;
391 }
